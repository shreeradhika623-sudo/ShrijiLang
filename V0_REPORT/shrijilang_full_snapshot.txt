#include <stdlib.h>
#include <string.h>
#include <stdio.h>
#include "env.h"

// Create environment
Env *new_env() {
    Env *env = (Env *)malloc(sizeof(Env));
    env->count = 0;
    return env;
}

// Set variable value
void env_set(Env *env, const char *name, int value) {

    // Check if variable exists
    for (int i = 0; i < env->count; i++) {
        if (strcmp(env->vars[i].name, name) == 0) {
            env->vars[i].value = value;
            return;
        }
    }

    // Add new variable
    strcpy(env->vars[env->count].name, name);
    env->vars[env->count].value = value;
    env->count++;
}

// Get variable value
int env_get(Env *env, const char *name) {
    for (int i = 0; i < env->count; i++) {
        if (strcmp(env->vars[i].name, name) == 0) {
            return env->vars[i].value;
        }
    }

    printf("'L Error: Variable '%s' not found\n", name);
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#include "include/ast.h"
#include "include/parser.h"
#include "include/interpreter.h"
#include "include/env.h"

int main() {

    printf("ShrijiLang (Sakhi + Niyu AI Enabled) Ready\n");
    printf("Type 'exit' to quit\n\n");

    // Create environment (variable storage)
    Env *env = new_env();

    char line[1024];

    while (1) {

        printf(">>> ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) {
            break;
        }

        if (strncmp(line, "exit", 4) == 0) {
            break;
        }

        // Parse the input into AST
        ASTNode *node = parse_program(line);

        if (!node) {
            printf("L Error: Cannot parse input\n");
            continue;
        }

        // Evaluate program
        eval(node, env);

        // No print_value() — because this function does not exist
        // AST output will be printed by Sakhi or Niyu automatically

        // Free memory
        ast_free(node);
    }

    return 0;
}

--- FILE: ./src/token.c ---
#include <ctype.h>
#include <string.h>
#include "../include/token.h"

// Global tokenizer source pointer
static const char *current;

// ----------------------------------------
// INITIALIZE TOKENIZER
// ----------------------------------------
void init_tokenizer(const char *src) {
    current = src;
}

// ----------------------------------------
// MAKE TOKEN
// ----------------------------------------
static Token make_token(TokenType type, const char *start, int length) {
    Token token;
    token.type = type;
    token.start = start;
    token.length = length;
    return token;
}

// ----------------------------------------
// MATCH KEYWORD
// ----------------------------------------
static TokenType check_keyword(const char *start, int length) {
    if (length == 4 && strncmp(start, "mavi", 4) == 0)
        return TOKEN_MAVI;

    if (length == 5 && strncmp(start, "sakhi", 5) == 0)
        return TOKEN_SAKHI;

    if (length == 4 && strncmp(start, "niyu", 4) == 0)
        return TOKEN_NIYU;
    return TOKEN_IDENTIFIER;
}

// ----------------------------------------
// NUMBER TOKEN
// ----------------------------------------
static Token number(const char *start) {
    while (isdigit(*current)) current++;
    return make_token(TOKEN_NUMBER, start, current - start);
}

// ----------------------------------------
// STRING TOKEN
// ----------------------------------------
static Token string(const char *start) {
    current++; // skip beginning quote
    while (*current != '"' && *current != '\0') current++;

    if (*current == '"') current++; // skip ending quote

    return make_token(TOKEN_STRING, start, current - start);
}

// ----------------------------------------
// IDENTIFIER OR KEYWORD
// ----------------------------------------
static Token identifier(const char *start) {
    while (isalnum(*current)) current++;

    return make_token(
        check_keyword(start, current - start),
        start,
        current - start
    );
}

// ----------------------------------------
// SCAN NEXT TOKEN
// ----------------------------------------
Token scan_token() {
    while (*current == ' ' || *current == '\t')
        current++;

    const char *start = current;

    if (*current == '\0') return make_token(TOKEN_EOF, start, 0);

    if (isdigit(*current)) return number(start);
    if (isalpha(*current)) return identifier(start);

    switch (*current) {
        case '+': current++; return make_token(TOKEN_PLUS, start, 1);
        case '-': current++; return make_token(TOKEN_MINUS, start, 1);
        case '*': current++; return make_token(TOKEN_STAR, start, 1);
        case '/': current++; return make_token(TOKEN_SLASH, start, 1);
        case '=': current++; return make_token(TOKEN_EQUAL, start, 1);
        case '(': current++; return make_token(TOKEN_LEFT_PAREN, start, 1);
        case ')': current++; return make_token(TOKEN_RIGHT_PAREN, start, 1);

        case '"': return string(start);

        default:
            current++;
            return make_token(TOKEN_ERROR, start, 1);
    }
}

#include <stdio.h>

// --------------------------------------
// CHAKRA ERROR SYSTEM (CES)
// --------------------------------------
void chakra_error(const char *msg, const char *chakra) {
    printf("\n🔮 Chakra Error (%s): %s\n\n", chakra, msg);
}


// --------------------------------------
// EVAL (MAIN EXECUTION ENGINE)
// --------------------------------------
int eval(ASTNode *node, Env *env) {

    if (node == NULL) {
        chakra_error("Empty expression cannot be evaluated.", "Mooladhara");
        return 0;
    }

    switch (node->type) {

    // ----------------------------
    // NUMBER
    // ----------------------------
    case AST_NUMBER:
        return node->number_value;


    // ----------------------------
    // IDENTIFIER
    // ----------------------------
    case AST_IDENTIFIER: {
        int value = env_get(env, node->name);
        return value;
    }


    // ----------------------------
    // ASSIGNMENT (mavi x = expr)
    // ----------------------------
    case AST_ASSIGNMENT: {
        int value = eval(node->value, env);
        env_set(env, node->name, value);
        return value;
    }


    // ----------------------------
    // BINARY EXPRESSIONS
    // ----------------------------
    case AST_BINARY: {

        int left  = eval(node->left, env);
        int right = eval(node->right, env);

        switch (node->op[0]) {

            case '+':
                return left + right;

            case '-':
                return left - right;

            case '*':
                return left * right;

            case '/':
                if (right == 0) {
                    chakra_error("Division by zero breaks clarity.", "Ajna");
                    return 0;
                }
                return left / right;

            default:
                chakra_error("Unknown operator used.", "Vishuddha");
                return 0;
        }
    }


    // ----------------------------
    // STRING LITERAL
    // ----------------------------
    case AST_STRING:
        printf("%s\n", node->string_value);
        return 0;


    // ----------------------------
    // SAKHI COMMAND
    // ----------------------------
    case AST_SAKHI: {

        if (!node->value || node->value->type != AST_STRING) {
            chakra_error("sakhi requires a string literal.", "Anahata");
            return 0;
        }

        sakhi_bridge(node->value->string_value);
        return 0;

    case AST_NIYU:
        niyu_bridge(node->value->string_value);
        return 0;
    }


    // ----------------------------
    // UNKNOWN NODE
    // ----------------------------
    default:
        chakra_error("Unknown AST node encountered.", "Sahasrara");
        return 0;
    }
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#include "../include/parser.h"
#include "../include/token.h"
#include "../include/ast.h"

//
// CURRENT TOKEN
//
static Token current;

//
// FORWARD DECLARATIONS
//
static ASTNode *parse_expression();
static ASTNode *parse_term();
static ASTNode *parse_factor();
static ASTNode *parse_assignment();
static ASTNode *parse_sakhi();
static ASTNode *parse_niyu();
static ASTNode *parse_statement();
//
// ADVANCE TOKEN
//
static void advance() {
    current = scan_token();
}

//
// MATCH TOKEN
//
static int match(TokenType type) {
    if (current.type == type) {
        advance();
        return 1;
    }
    return 0;
}

//
// EXPECT TOKEN
//
static void expect(TokenType type, const char *msg) {
    if (!match(type)) {
        printf("L Error: %s\n", msg);
    }
}

//
// PROGRAM ENTRY
//
ASTNode *parse_program(const char *source) {
    init_tokenizer(source);
    advance();
    return parse_statement();
}

//
// STATEMENT
//
ASTNode *parse_statement() {

    // sakhi "message"
    if (current.type == TOKEN_SAKHI)
        return parse_sakhi();

    // niyu "message"
    if (current.type == TOKEN_NIYU)
        return parse_niyu();

    // mavi x = expr
    if (current.type == TOKEN_MAVI) {
        advance();
        return parse_assignment();
    }

    // default = expression
    return parse_expression();
}

//
// PARSE SAKHI
//
static ASTNode *parse_sakhi() {
    advance(); // skip sakhi

    if (current.type != TOKEN_STRING) {
        printf("L Error: sakhi expects a string literal\n");
        return NULL;
    }

    char buffer[256] = {0};
    strncpy(buffer, current.start + 1, current.length - 2);

    advance();
    return new_sakhi_node(new_string_node(buffer));
}

//
// PARSE NIYU
//
static ASTNode *parse_niyu() {
    advance(); // skip niyu

    if (current.type != TOKEN_STRING) {
        printf("L Error: niyu expects a string literal\n");
        return NULL;
    }

    char buffer[256] = {0};
    strncpy(buffer, current.start + 1, current.length - 2);

    advance();
    return new_niyu_node(new_string_node(buffer));
}

//
// ASSIGNMENT
// mavi x = expression
//
static ASTNode *parse_assignment() {

    if (current.type != TOKEN_IDENTIFIER) {
        printf("L Error: Expected variable name after 'mavi'\n");
        return NULL;
    }

    char name[128] = {0};
    strncpy(name, current.start, current.length);
    advance();

    expect(TOKEN_EQUAL, "Expected '=' in assignment");

    ASTNode *value = parse_expression();
    return new_assignment_node(name, value);
}

//
// EXPRESSION ( + , - )
//
static ASTNode *parse_expression() {
    ASTNode *node = parse_term();

    while (current.type == TOKEN_PLUS || current.type == TOKEN_MINUS) {
        char op = (current.type == TOKEN_PLUS) ? '+' : '-';
        advance();

        ASTNode *right = parse_term();
        node = new_binary_node(op, node, right);
    }

    return node;
}

//
// TERM ( * , / )
//
static ASTNode *parse_term() {
    ASTNode *node = parse_factor();

    while (current.type == TOKEN_STAR || current.type == TOKEN_SLASH) {
        char op = (current.type == TOKEN_STAR) ? '*' : '/';
        advance();

        ASTNode *right = parse_factor();
        node = new_binary_node(op, node, right);
    }

    return node;
}

//
// FACTOR
// NUMBER | STRING | IDENTIFIER | "(" expr ")"
//
static ASTNode *parse_factor() {

    // number
    if (current.type == TOKEN_NUMBER) {
        int value = atoi(current.start);
        advance();
        return new_number_node(value);
    }

    // string
    if (current.type == TOKEN_STRING) {
        char buffer[256] = {0};
        strncpy(buffer, current.start + 1, current.length - 2);
        advance();
        return new_string_node(buffer);
    }

    // identifier
    if (current.type == TOKEN_IDENTIFIER) {
        char name[128] = {0};
        strncpy(name, current.start, current.length);
        advance();
        return new_identifier_node(name);
    }

    // ( expr )
    if (match(TOKEN_LEFT_PAREN)) {
        ASTNode *node = parse_expression();
        expect(TOKEN_RIGHT_PAREN, "Missing ')'");
        return node;
    }

    printf("L Error: Unexpected token\n");
    return NULL;
}

static void safe_copy(char *dest, const char *src, int max) {
    if (!src) {
        dest[0] = '\0';
        return;

    strncpy(dest, src, max - 1);
    dest[max - 1] = '\0';

void env_set(Env *env, const char *name, int value) {
    // check if exists
    for (int i = 0; i < env->count; i++) {
        if (strcmp(env->vars[i].name, name) == 0) {
            env->vars[i].value = value;
            return;
        }
    }

    // new variable
    strcpy(env->vars[env->count].name, name);
    env->vars[env->count].value = value;
    env->count++;
}

int env_get(Env *env, const char *name) {
    for (int i = 0; i < env->count; i++) {
        if (strcmp(env->vars[i].name, name) == 0) {
            return env->vars[i].value;
        }
    }

    printf("Runtime Error: variable '%s' not found.\n", name);
    exit(1);
}

        if (strcmp(last_messages[i], msg) == 0) {
            printf("[Memory📚] Pattern: Repeated thought detected.\n");

        }
    }

    // Save message
    strncpy(last_messages[msg_index], msg, 255);
    last_messages[msg_index][255] = '\0';

    // Update counters safely
    msg_index = (msg_index + 1) % 5;
    if (stored_count < 5) stored_count++;

    // Stress detection
    if (strstr(msg, "can't") || strstr(msg, "not possible") || strstr(msg, "stuck")) {

    }
}

        return NULL;

    int index = (msg_index - 1 + 5) % 5;

//   - Gives FINAL INTERPRETATION of user's mental state

void niyu_synthesis(const char *msg) {

    printf("[Synth🧬] Synthesizing understanding...\n");

    int has_negative = contains_word(msg, "fail") ||
                       contains_word(msg, "stuck") ||
                       contains_word(msg, "can't");

    int has_positive = contains_word(msg, "improve") ||
                       contains_word(msg, "progress") ||
                       contains_word(msg, "better");

    int desire = contains_word(msg, "want") ||
                 contains_word(msg, "wish") ||
                 contains_word(msg, "hope");

    int no_action = contains_word(msg, "nothing") ||
                    contains_word(msg, "do nothing");

    // ------------------------------------------------------
    // FINAL SYNTHESIS LOGIC
    // ------------------------------------------------------

    if (has_negative && has_positive) {
        printf("[Synth🧬] Mixed emotional state detected.\n");
        printf("[Synth🧬] Interpretation: User is improving but still fighting doubt.\n");
        return;

        printf("[Synth🧬] Desire without action detected.\n");
        printf("[Synth🧬] Interpretation: User wants change but is stuck in inaction.\n");
        return;
    }

    // Case 3 — Strong contradiction
    if (contradiction) {
        printf("[Synth🧬] Contradiction present → Message needs resolution.\n");
        printf("[Synth🧬] Interpretation: User holds two opposing thoughts.\n");
        return;

        printf("[Synth🧬] Negative-only pattern detected.\n");
        printf("[Synth🧬] Interpretation: User is feeling low or discouraged.\n");

    if (has_positive && !has_negative) {
        printf("[Synth🧬] Positive growth signal detected.\n");
        printf("[Synth🧬] Interpretation: User is genuinely improving.\n");
        return;
    }

    // Case 6 — Neutral
    printf("[Synth🧬] Stable message.\n");
    printf("[Synth🧬] Interpretation: No emotional or logical conflict detected.\n");

    if (strstr(msg, "improve") && strstr(last_msg, "failed"))

           strstr(msg, "stuck") ||
           strstr(msg, "can't") ||
           strstr(msg, "nothing");
}

// -------------------------------------------

// -------------------------------------------
int is_positive_event(const char *msg) {
    return strstr(msg, "improve") ||
           strstr(msg, "improving") || 
           strstr(msg, "progress") ||
           strstr(msg, "better") ||
           strstr(msg, "learning");
}

// -------------------------------------------
// MAIN INDUCTIVE ENGINE
// -------------------------------------------
void niyu_induce(const char *msg) {

    printf("[Induce🔮] Inductive reasoning active...\n");

    int neg_flag = is_negative_event(msg);
    int pos_flag = is_positive_event(msg);

    // 1. Count patterns
    if (neg_flag) {
        count_negative++;
        printf("[Induce🔮] Negative pattern +1 (Total: %d)\n", count_negative);
    }

    if (pos_flag) {
        count_positive++;
        printf("[Induce🔮] Positive pattern +1 (Total: %d)\n", count_positive);
    }

    // 2. Repeated negative events
    if (count_negative >= 2) {
        printf("[Induce🔮] Pattern: Repeated negative events → Likely future issue.\n");
    }

    // 3. Repeated positive events
    if (count_positive >= 2) {
        printf("[Induce🔮] Pattern: Improvement trend detected.\n");
    }

    // 4. Mixed pattern detection
    if (count_negative >= 1 && count_positive >= 1) {
        printf("[Induce🔮] Mixed pattern: User is fluctuating between negative and positive states.\n");
    }

    // 5. No informative pattern
    if (!neg_flag && !pos_flag) {
        printf("[Induce🔮] No pattern detected.\n");
    }
