/*──────────────────────────────────────────────────────────────
 | SHRIJILANG — MAIN ENTRY
 |
 | Modes:
 |   1) REPL        → ./shrijilang
 |   2) SCRIPT RUN  → ./shrijilang run file.sri
 |
 | Core rule:
 |   • .sri file is parsed as ONE PROGRAM (AST_PROGRAM)
 |   • No line-by-line execution
 *──────────────────────────────────────────────────────────────*/

#define INPUT_MAX 1024
#define FILE_BUFFER_MAX 8192

/*──────────────────────────────────────────────────────────────
 | Banner
 *──────────────────────────────────────────────────────────────*/
static void print_banner(void) {
    printf("──────────────────────────────────────────────\n");
    printf("        ShrijiLang — Universe-Class V1\n");
    printf("   (Sakhi + Niyu + Shiri + Mira + Kavya AI)\n");
    printf("──────────────────────────────────────────────\n");
}

/*──────────────────────────────────────────────────────────────
 | Run .sri script as FULL PROGRAM  ✅ FIXED
 *──────────────────────────────────────────────────────────────*/
static int run_script(const char *filename) {

    FILE *fp = fopen(filename, "r");
    if (!fp) {
        perror("Unable to open .sri file");
        return 1;
    }

    Env *env = new_env();
    if (!env) {
        fclose(fp);
        return 1;
    }

    /* Read entire file into buffer */
    char buffer[FILE_BUFFER_MAX];
    buffer[0] = '\0';

    char line[INPUT_MAX];

    while (fgets(line, sizeof(line), fp)) {
        if (strlen(buffer) + strlen(line) >= FILE_BUFFER_MAX - 1) {
            fprintf(stderr, "Script too large\n");
            fclose(fp);
            return 1;
        }
        strcat(buffer, line);
    }

    fclose(fp);

    /* Parse full program */
    ASTNode *program = parse_program(buffer);
    if (!program) {
        /* parser already reported error */
        return 1;
    }

    /* Execute full program */
    run_program(program, env);

    ast_free(program);
    return 0;
}

/*──────────────────────────────────────────────────────────────
 | Interactive REPL
 *──────────────────────────────────────────────────────────────*/
static void run_repl(void) {

    Env *env = new_env();
    if (!env)
        return;

    char line[INPUT_MAX];

    while (1) {

        printf("shiri >>> ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin))
            break;

        line[strcspn(line, "\n")] = '\0';

        if (strcmp(line, "exit") == 0)
            break;

        if (line[0] == '\0')
            continue;

        ASTNode *node = parse_program(line);
        if (!node)
            continue;

        long long result = eval(node, env);

        /* Pure expression output (not assignment / identifier / command) */
        if (node->type != AST_IDENTIFIER &&
            node->type != AST_ASSIGNMENT &&
            node->type != AST_COMMAND) {

            char msg[128];
            snprintf(msg, sizeof(msg),
                     "Calculation result is %lld", result);

            const char *out = shiri_process(msg);
            if (out && *out)
                printf("%s\n", out);
        }

        ast_free(node);
    }
}

/*──────────────────────────────────────────────────────────────
 | MAIN
 *──────────────────────────────────────────────────────────────*/
int main(int argc, char *argv[]) {

    print_banner();

    /* 1) ./shrijilang file.sri  */
    if (argc == 2) {
        return run_script(argv[1]);
    }

    /* 2) ./shrijilang run file.sri (legacy) */
    if (argc == 3 && strcmp(argv[1], "run") == 0) {
        return run_script(argv[2]);
    }

    /* 3) REPL mode */
    run_repl();

    printf("\nShrijiLang session ended.\n");
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#include "../include/interpreter.h"
#include "../include/parser.h"
#include "../include/env.h"

// Global Environment
Env *GLOBAL_ENV;

/* Read entire file into memory */
static char *read_file(const char *path) {
    FILE *f = fopen(path, "rb");
    if (!f) {
        perror("File open failed");
        return NULL;
    }

    fseek(f, 0, SEEK_END);
    long size = ftell(f);
    rewind(f);

    char *buffer = (char *)malloc(size + 1);
    if (!buffer) {
        fclose(f);
        return NULL;
    }

    fread(buffer, 1, size, f);
    buffer[size] = '\0';
    fclose(f);
    return buffer;
}

static void run_script(const char *filename) {
    char *source = read_file(filename);
    if (!source) {
        printf("⛔ Failed to read %s\n", filename);
        return;
    }

    ASTNode *root = parse_program(source);
    if (!root) {
        free(source);
        return;
    }

    eval(root, GLOBAL_ENV);

    ast_free(root);
    free(source);
}

static void interactive_shell() {
    char line[512];

    printf("──────────────────────────────────────────────\n");
    printf("        🌸 ShrijiLang Interactive Shell 🌸\n");
    printf("──────────────────────────────────────────────\n");
    printf("Type 'exit' to quit\n\n");

    while (1) {
        printf("Shiri> ");
        fflush(stdout);

        if (!fgets(line, sizeof(line), stdin)) {
            printf("\n");
            break;
        }

        line[strcspn(line, "\n")] = 0;

        if (strcmp(line, "exit") == 0) {
            printf("Radhe Radhe 🌸\n");
            break;
        }

        if (line[0] == '\0')
            continue;

        ASTNode *root = parse_program(line);
        if (!root)
            continue;

        eval(root, GLOBAL_ENV);
        ast_free(root);
    }
}

int main(int argc, char **argv) {
    GLOBAL_ENV = new_env();

    if (argc > 1) {
        run_script(argv[1]);
    } else {
        interactive_shell();
    }

    return 0;
}

----- src/token.c -----
     1	#include <ctype.h>
     2	#include <string.h>
     3	#include <stdlib.h>
     4	
     5	#include "../include/token.h"
     6	
     7	/*──────────────────────────────────────────────
     8	  SHRIJILANG LEXER — FINAL STABLE (NEWLINE AWARE)
     9	──────────────────────────────────────────────*/
    10	
    11	static const char *start;
    12	static const char *current;
    13	
    14	/*──────── helpers ────────*/
    15	static int is_at_end(void) {
    16	    return *current == '\0';
    17	}
    18	
    19	static char advance_char(void) {
    20	    return *current++;
    21	}
    22	
    23	static char peek(void) {
    24	    return *current;
    25	}
    26	
    27	static char peek_next(void) {
    28	    if (is_at_end()) return '\0';
    29	    return current[1];
    30	}
    31	
    32	static Token make_token(TokenType type, const char *s, int len) {
    33	    Token t;
    34	    t.type = type;
    35	    t.start = s;
    36	    t.length = len;
    37	    return t;
    38	}
    39	
    40	/*──────── skip whitespace (NOT newline) ────────*/
    41	static void skip_ws(void) {
    42	    for (;;) {
    43	        char c = peek();
    44	        if (c==' ' || c=='\r' || c=='\t') {
    45	            advance_char();
    46	        } else {
    47	            return;
    48	        }
    49	    }
    50	}
    51	
    52	/*──────── keywords ────────*/
    53	static TokenType check_keyword(const char *s, int len) {
    54	    if (len==4 && strncmp(s,"mavi",4)==0) return TOKEN_MAVI;
    55	    if (len==4 && strncmp(s,"agar",4)==0) return TOKEN_AGAR;
    56	    if (len==5 && strncmp(s,"warna",5)==0) return TOKEN_WARNA;
    57	    if (len==6 && strncmp(s,"jabtak",6)==0) return TOKEN_JABTAK;
    58	    if (len==4 && strncmp(s,"nahi",4)==0) return TOKEN_NAHI;
    59	    if (len==4 && strncmp(s,"true",4)==0) return TOKEN_TRUE;
    60	    if (len==5 && strncmp(s,"false",5)==0) return TOKEN_FALSE;
    61	    return TOKEN_IDENTIFIER;
    62	}
    63	
    64	/*──────── identifiers ────────*/
    65	static Token identifier(void) {
    66	    while (isalnum(peek()) || peek()=='_') advance_char();
    67	    int len = (int)(current - start);
    68	    return make_token(check_keyword(start, len), start, len);
    69	}
    70	
    71	/*──────── numbers ────────*/
    72	static Token number(void) {
    73	    while (isdigit(peek())) advance_char();
    74	    int len = (int)(current - start);
    75	    return make_token(TOKEN_NUMBER, start, len);
    76	}
    77	
    78	/*──────── string ────────*/
    79	static Token string(void) {
    80	    while (peek()!='"' && !is_at_end()) advance_char();
    81	    if (!is_at_end()) advance_char();   // closing "
    82	    int len = (int)(current - start);
    83	    return make_token(TOKEN_STRING, start, len);
    84	}
    85	
    86	/*──────── public API ────────*/
    87	void init_tokenizer(const char *src) {
    88	    start = src;
    89	    current = src;
    90	}
    91	
    92	Token scan_token(void) {
    93	    skip_ws();
    94	    start = current;
    95	
    96	    if (is_at_end()) return make_token(TOKEN_EOF, start, 0);
    97	
    98	    /* NEWLINE token */
    99	    if (peek() == '\n') {
   100	        advance_char();
   101	        return make_token(TOKEN_NEWLINE, start, 1);
   102	    }
   103	
   104	    char c = advance_char();
   105	
   106	    if (isalpha(c) || c=='_') return identifier();
   107	    if (isdigit(c)) return number();
   108	
   109	    switch (c) {
   110	        case '(': return make_token(TOKEN_LEFT_PAREN, start, 1);
   111	        case ')': return make_token(TOKEN_RIGHT_PAREN, start, 1);
   112	        case '{': return make_token(TOKEN_LEFT_BRACE, start, 1);
   113	        case '}': return make_token(TOKEN_RIGHT_BRACE, start, 1);
   114	
   115	        case '+': return make_token(TOKEN_PLUS, start, 1);
   116	        case '-': return make_token(TOKEN_MINUS, start, 1);
   117	        case '*': return make_token(TOKEN_STAR, start, 1);
   118	        case '/': return make_token(TOKEN_SLASH, start, 1);
   119	        case '%': return make_token(TOKEN_MOD, start, 1);
   120	
   121	        case '=':
   122	            if (peek_next()=='=') {
   123	                advance_char();
   124	                return make_token(TOKEN_EQEQ, start, 2);
   125	            }
   126	            return make_token(TOKEN_EQUAL, start, 1);
   127	
   128	        case '!':
   129	            if (peek_next()=='=') {
   130	                advance_char();
   131	                return make_token(TOKEN_NEQ, start, 2);
   132	            }
   133	            break;
   134	
   135	        case '>':
   136	            if (peek_next()=='=') {
   137	                advance_char();
   138	                return make_token(TOKEN_GTE, start, 2);
   139	            }
   140	            return make_token(TOKEN_GT, start, 1);
   141	
   142	        case '<':
   143	            if (peek_next()=='=') {
   144	                advance_char();
   145	                return make_token(TOKEN_LTE, start, 2);
   146	            }
   147	            return make_token(TOKEN_LT, start, 1);
   148	
   149	        case '"':
   150	            return string();
   151	    }
   152	
   153	    return make_token(TOKEN_ERROR, start, 1);
   154	}

   239	          (current.type == TOKEN_LT)   ? '<' :

   271

   273	        return NULL;   // stop expression before block
   274	
   275	    if (current.type == TOKEN_NUMBER){
   276	        int v = atoi(current.start);
   277	        advance();
   278	        return new_number_node(v);
   279	    }
   280	
   281	    if (current.type == TOKEN_TRUE){
   282	        advance();
   283	        return new_number_node(1);
   284	    }
   285	
   286	    if (current.type == TOKEN_FALSE){
   287	        advance();
   288	        return new_number_node(0);
   289	    }
   290	
   291	    if (current.type == TOKEN_IDENTIFIER){
   292	        char name[128];
   293	        strncpy(name, current.start, current.length);
   294	        name[current.length] = 0;
   295	        advance();
   296	        return new_identifier_node(name);
   297	    }
   298	
   299	    if (current.type == TOKEN_LEFT_PAREN){
   300	        advance();
   301	        ASTNode *n = parse_expression();
   302	        expect(TOKEN_RIGHT_PAREN, E_PARSE_02, ")", "missing )", "(expr)");
   303	        return n;
   304	    }
   305	
   306	    shriji_error(E_PARSE_02, "expr", "bad token", "check syntax");
   307	    return NULL;
   308	}

    45	    if (b == 0) {

    60	/*──────────────────────────────────────────────────────────────
    61	 | MAIN EVALUATOR
    62	 *──────────────────────────────────────────────────────────────*/
    63	long long eval(ASTNode *node, Env *env) {
    64	
    65	    static int state_initialized = 0;
    66	    if (!state_initialized) {
    67	        state_init(&SHRIJI_STATE);
    68	        state_initialized = 1;
    69	    }
    70	
    71	    /* 🔐 Global safety gate */
    72	    if (!state_allows_execution())
    73	        return 0;
    74	
    75	    if (!node)
    76	        return 0;
    77	
    78	    /* 🧠 STEP COUNTER (runtime guard) */
    79	    SHRIJI_STATE.steps_used++;
    80	
    81	    if (SHRIJI_STATE.steps_used > SHRIJI_STATE.max_steps) {
    82	        shriji_error(
    83	            E_PARSE_02,
    84	            "runtime",
    85	            "execution limit exceeded",
    86	            "possible infinite loop detected"
    87	        );
    88	        SHRIJI_STATE.safety = STATE_CRITICAL;
    89	        return 0;
    90	    }
    91	
    92	    switch (node->type) {
    93	
    94	    /*──────── NUMBER ────────*/
    95	    case AST_NUMBER:
    96	        return node->number_value;
    97	
    98	    /*──────── IDENTIFIER ────────*/
    99	    case AST_IDENTIFIER: {
   100	
   101	        if (!env_exists(env, node->name)) {
   102	
   103	            char event[64];
   104	            snprintf(event, sizeof(event),
   105	                     "shunya:%.*s", 48, node->name);
   106	
   107	            const char *out = shiri_process(event);
   108	            if (out && *out)
   109	                printf("%s\n", out);
   110	
   111	            return 0;
   112	        }
   113	        return env_get(env, node->name);
   114	    }
   115	
   116	    /*──────── ASSIGNMENT ────────*/
   117	    case AST_ASSIGNMENT: {
   118	
   119	        long long value = eval(node->value, env);
   120	        env_set(env, node->name, value);
   121	
   122	        event_fire(EVENT_ASSIGNMENT, node->name);
   123	        state_on_success(&SHRIJI_STATE);
   124	
   125	        strncpy(
   126	            SHRIJI_DIARY.last_var,
   127	            node->name,
   128	            sizeof(SHRIJI_DIARY.last_var) - 1
   129	        );
   130	        SHRIJI_DIARY.last_var[sizeof(SHRIJI_DIARY.last_var) - 1] = '\0';
   131	
   132	        SHRIJI_DIARY.last_value = value;
   133	        SHRIJI_DIARY.has_assignment = 1;
   134	
   135	        char msg[64];
   136	        snprintf(msg, sizeof(msg), "result:%lld", value);
   137	
   138	        const char *out = shiri_process(msg);
   139	        if (out && *out)
   140	            printf("%s\n", out);
   141	
   142	        return value;
   143	    }
   144	
   145	    /*──────── UPDATE (x = 10) ────────*/
   146	    case AST_UPDATE: {
   147	
   148	        if (!env_exists(env, node->name)) {
   149	            shriji_error(
   150	                E_ASSIGN_01,
   151	                node->name,
   152	                "variable not declared",
   153	                "use mavi x = ... first"
   154	            );
   155	            return 0;
   156	        }
   157	
   158	        long long value = eval(node->value, env);
   159	        env_set(env, node->name, value);
   160	
   161	        char msg[64];
   162	        snprintf(msg, sizeof(msg), "update:%lld", value);
   163	
   164	        const char *out = shiri_process(msg);
   165	        if (out && *out)
   166	            printf("%s\n", out);
   167	
   168	        return value;
   169	    }
   170	
   171	    /*──────── BINARY ────────*/
   172	    case AST_BINARY: {
   173	
   174	        long long L = eval(node->left, env);
   175	        long long R = eval(node->right, env);
   176	
   177	        switch (node->op[0]) {
   178	            case '+': return L + R;
   179	            case '-': return L - R;
   180	            case '*': return L * R;
   181	            case '/': return safe_div(L, R);
   182	            case '>': return (L > R);
   183	            case '<': return (L < R);
   184	            case '=': return (L == R);
   185	            case '!': return (L != R);
   186	            default:
   187	                shriji_error(
   188	                    E_PARSE_02,
   189	                    "binary",
   190	                    "unknown operator",
   191	                    "check expression"
   192	                );
   193	                return 0;
   194	        }
   195	    }
   196	
   197	    /*──────── LOGICAL NOT ────────*/
   198	    case AST_NOT: {
   199	        long long v = eval(node->left, env);
   200	        return !v;
   201	    }
   202	
   203	case AST_PROGRAM: {
   204	    long long last = 0;
   205	    for (int i = 0; i < node->stmt_count; i++) {
   206	        last = eval(node->statements[i], env);
   207	    }
   208	    return last;
   209	}
   210	
   211	    /*──────── BLOCK ────────*/
   212	    case AST_BLOCK: {
   213	        long long last = 0;
   214	        for (int i = 0; i < node->stmt_count; i++) {
   215	            last = eval(node->statements[i], env);
   216	        }
   217	        return last;
   218	    }
   219	
   220	    /*──────── IF / WARNA ────────*/
   221	    case AST_IF: {
   222	
   223	        long long cond = eval(node->condition, env);
   224	
   225	        if (cond) {
   226	            return eval(node->left, env);
   227	        } else if (node->else_block) {
   228	            return eval(node->else_block, env);
   229	        }
   230	        return 0;
   231	    }
   232	
   233	    /*──────── COMMAND ────────*/
   234	    case AST_COMMAND: {
   235	
   236	        CommandType cmd = resolve_command(node->command_name);
   237	
   238	        if (cmd == CMD_UNKNOWN) {
   239	            shriji_error(
   240	                E_PARSE_02,
   241	                "command",
   242	                "unknown command",
   243	                "command not recognized"
   244	            );
   245	            return 0;
   246	        }
   247	
   248	        int status = execute_command(cmd);
   249	
   250	        char msg[128];
   251	        snprintf(
   252	            msg,
   253	            sizeof(msg),
   254	            "Command executed internally as %s",
   255	            command_name(cmd)
   256	        );
   257	
   258	        const char *out = shiri_process(msg);
   259	        if (out && *out)
   260	            printf("%s\n", out);
   261	
   262	        return status;
   263	    }
   264	
   265	    default:
   266	        return 0;
   267	    }
   268	}
   269	/*──────────────────────────────────────────────────────────────
   270	 | PROGRAM RUNNER
   271	 *──────────────────────────────────────────────────────────────*/

   273	
   274	    if (!program || program->type != AST_PROGRAM)
   275	        return 0;
   276	
   277	    long long last = 0;
   278	
   279	    for (int i = 0; i < program->stmt_count; i++) {
   280	        last = eval(program->statements[i], env);
   281	    }
   282	    return last;

    63	        sizeof(env->vars[env->count].name) - 1
